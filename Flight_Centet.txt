CREATE TABLE flights
  (
      id serial unique ,
      airline_company_id bigint,
      origin_country_id bigint,
      destination_country_id bigint,
      departure_time date,
      landing_time date,
      remaining_tickets int,
      PRIMARY KEY (id),
      FOREIGN KEY (airline_company_id) references airline_companies(id),
      FOREIGN KEY (origin_country_id) references countries(id),
      FOREIGN KEY (destination_country_id) references countries (id)
);

CREATE TABLE countries
  (
      id serial unique ,
      name text unique ,
      PRIMARY KEY (id)
);

CREATE TABLE airline_companies
  (
      id serial unique ,
      name text unique ,
      country_id bigint,
      user_id bigint unique ,
      PRIMARY KEY (id),
      FOREIGN KEY (country_id) REFERENCES countries (id),
      FOREIGN KEY (user_id) REFERENCES users (id)
);

CREATE TABLE users
  (
      id serial unique ,
      username text unique ,
      user_role bigint,
      password text,
      email text unique ,
      PRIMARY KEY (id),
      FOREIGN KEY (user_role) REFERENCES user_roles (id)
);

CREATE TABLE user_roles
  (
      id serial unique ,
      role_mame text unique
);

CREATE TABLE administrators
  (
      id serial unique ,
      first_name text,
      last_name bigint,
      level int ,
      user_id bigint,
      PRIMARY KEY (id),
      FOREIGN KEY (user_id) REFERENCES users (id)
);

CREATE TABLE customers
  (
      id serial unique ,
      first_name text,
      last_name bigint,
      address text,
      phone_no text unique ,
      credit_card_no text unique ,
      user_id bigint unique ,
      PRIMARY KEY (id),
      FOREIGN KEY (user_id) REFERENCES users (id)
);

CREATE TABLE tickets
  (
      id serial unique ,
      flight_id bigint,
      customer_id bigint,
      UNIQUE (flight_id, customer_id),
      PRIMARY KEY (id),
      FOREIGN KEY (flight_id) REFERENCES flights (id),
      FOREIGN KEY (customer_id) REFERENCES customers (id)
);
